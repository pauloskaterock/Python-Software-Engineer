# 41. Mover mouse com verificação de failsafe
def safe_move_to(x, y, duration=0.5):
    """Move o mouse com segurança"""
    if pyautogui.FAILSAFE and (x == 0 and y == 0):
        raise pyautogui.FailSafeException("Failsafe ativado!")
    pyautogui.moveTo(x, y, duration=duration)

# 42. Clicar com verificação de visibilidade
def click_if_visible(image_path, confidence=0.8, timeout=5):
    """Clica em imagem se estiver visível"""
    start_time = time.time()
    while time.time() - start_time < timeout:
        try:
            location = pyautogui.locateOnScreen(image_path, confidence=confidence)
            if location:
                center = pyautogui.center(location)
                pyautogui.click(center)
                return True
        except pyautogui.ImageNotFoundException:
            time.sleep(0.5)
    return False

# 44. Esperar por mudança de cor
def wait_for_color(x, y, expected_color, timeout=10, interval=0.5):
    """Espera até que um pixel tenha a cor esperada"""
    start_time = time.time()
    while time.time() - start_time < timeout:
        current_color = pyautogui.pixel(x, y)
        if current_color == expected_color:
            return True
        time.sleep(interval)
    return False

def mouse_position_monitor(interval=1, duration=10):
    """Monitora a posição do mouse por um tempo"""
    start_time = time.time()
    positions = []

    print("Iniciando monitoramento de mouse...")
    print("Mova o mouse para o canto superior esquerdo para parar.")

    try:
        while time.time() - start_time < duration:
            x, y = pyautogui.position()
            positions.append((x, y, time.time()))
            print(f"Posição: ({x}, {y})")
            time.sleep(interval)
    except KeyboardInterrupt:
        print("\nMonitoramento interrompido pelo usuário.")

    return positions

"""
EXEMPLO COMPLETO: Automação de Login e Exportação
Automação RPA usando apenas PyAutoGUI
"""

import pyautogui
import time
import logging
from datetime import datetime

# Configuração
pyautogui.FAILSAFE = True
pyautogui.PAUSE = 0.3

# Configurar logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

class RPAPyAutoGUI:
    def __init__(self):
        self.screen_width, self.screen_height = pyautogui.size()
        logger.info(f"Resolução da tela: {self.screen_width}x{self.screen_height}")

    def take_screenshot(self, name="screenshot"):
        """Tira screenshot e salva com timestamp"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"{name}_{timestamp}.png"
        pyautogui.screenshot(filename)
        logger.info(f"Screenshot salvo: {filename}")
        return filename

    def click_image(self, image_path, timeout=10, confidence=0.9):
        """Clica em uma imagem quando ela aparece"""
        logger.info(f"Procurando imagem: {image_path}")

        start_time = time.time()
        while time.time() - start_time < timeout:
            try:
                location = pyautogui.locateOnScreen(image_path, confidence=confidence)
                if location:
                    center = pyautogui.center(location)
                    pyautogui.click(center)
                    logger.info(f"Imagem encontrada e clicada em {center}")
                    return True
            except pyautogui.ImageNotFoundException:
                time.sleep(1)

        logger.warning(f"Imagem não encontrada em {timeout} segundos: {image_path}")
        return False

    def fill_field(self, x, y, text, clear_first=True):
        """Preenche campo em coordenadas específicas"""
        pyautogui.click(x, y)
        if clear_first:
            pyautogui.hotkey('ctrl', 'a')
            pyautogui.press('backspace')
        pyautogui.write(text)
        logger.info(f"Campo preenchido: {text}")

    def login_system(self, username, password):
        """Automação de login"""
        logger.info("Iniciando processo de login")

        # Abrir sistema (exemplo)
        pyautogui.hotkey('winleft', 'r')
        time.sleep(1)
        pyautogui.write('https://sistema.com')
        pyautogui.press('enter')
        time.sleep(3)

        # Preencher credenciais
        self.fill_field(300, 250, username)  # Campo usuário
        pyautogui.press('tab')
        pyautogui.write(password)  # Campo senha
        pyautogui.press('enter')

        # Verificar login
        time.sleep(2)
        self.take_screenshot("apos_login")

        logger.info("Login concluído")

    def export_data(self, export_path):
        """Exporta dados do sistema"""
        logger.info("Iniciando exportação de dados")

        # Navegar para menu de exportação
        self.click_image('images/menu_exportar.png')
        time.sleep(1)

        # Selecionar formato
        pyautogui.click(400, 300)  # Clique em Excel
        time.sleep(0.5)

        # Clicar em exportar
        pyautogui.click(500, 400)
        time.sleep(2)

        # Salvar diálogo
        pyautogui.write(export_path)
        pyautogui.press('enter')

        # Aguardar exportação
        if self.wait_for_export_complete():
            logger.info("Exportação concluída com sucesso")
            return True
        else:
            logger.error("Falha na exportação")
            return False

    def wait_for_export_complete(self, timeout=30):
        """Aguarda até que a exportação esteja completa"""
        logger.info("Aguardando conclusão da exportação...")

        # Monitorar pixel que indica conclusão
        complete_color = (0, 128, 0)  # Verde
        start_time = time.time()

        while time.time() - start_time < timeout:
            # Verificar cor do pixel de status
            if pyautogui.pixelMatchesColor(600, 500, complete_color, tolerance=10):
                logger.info("Exportação detectada como completa")
                return True

            # Verificar se há mensagem de erro
            if pyautogui.pixelMatchesColor(600, 500, (255, 0, 0), tolerance=10):  # Vermelho
                logger.error("Erro detectado durante exportação")
                return False

            time.sleep(1)

        logger.warning(f"Timeout após {timeout} segundos")
        return False

    def run_full_process(self):
        """Executa processo RPA completo"""
        try:
            logger.info("=" * 50)
            logger.info("INICIANDO PROCESSO RPA")
            logger.info("=" * 50)

            # Notificar usuário
            pyautogui.alert(
                "O processo RPA será iniciado.\n"
                "Não use o mouse ou teclado durante a execução.\n"
                "Para abortar, mova o mouse para o canto superior esquerdo.",
                "Sistema RPA"
            )

            # Capturar estado inicial
            self.take_screenshot("inicio_processo")

            # Executar passos
            self.login_system("admin", "senha123")
            time.sleep(2)

            self.export_data("C:\\Exportados\\dados.xlsx")
            time.sleep(2)

            # Finalizar
            self.take_screenshot("fim_processo")

            pyautogui.alert("Processo RPA concluído com sucesso!", "Sistema RPA")
            logger.info("Processo concluído com sucesso")

        except pyautogui.FailSafeException:
            logger.warning("Processo abortado pelo usuário (failsafe)")
            pyautogui.alert("Processo RPA abortado pelo usuário", "Sistema RPA")

        except Exception as e:
            logger.error(f"Erro durante o processo: {str(e)}")
            self.take_screenshot("erro_processo")
            pyautogui.alert(f"Erro no processo RPA: {str(e)[:50]}", "Sistema RPA")

# Executar
if __name__ == "__main__":
    rpa = RPAPyAutoGUI()
    rpa.run_full_process()

# 1. Calibrar coordenadas dinamicamente
def get_center_of_screen():
    width, height = pyautogui.size()
    return width // 2, height // 2

# 2. Encontrar elemento por cor
def find_by_color(target_color, region=None):
    screenshot = pyautogui.screenshot(region=region)
    width, height = screenshot.size

    for x in range(width):
        for y in range(height):
            if screenshot.getpixel((x, y)) == target_color:
                return x, y
    return None

# 3. Automatizar arrastar arquivo
def drag_file_to_folder(file_coords, folder_coords):
    pyautogui.moveTo(file_coords)
    pyautogui.mouseDown()
    pyautogui.moveTo(folder_coords, duration=1)
    pyautogui.mouseUp()
