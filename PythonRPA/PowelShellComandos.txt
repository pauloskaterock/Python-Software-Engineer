$BackupDir = "C:\Backups"

# Função de logging
function Write-Log {
    param(
        [string]$Message,
        [string]$Level = "INFO"
    )
    $timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $logEntry = "$timestamp [$Level] $Message"
    Write-Host $logEntry
    $logEntry | Out-File $LogFile -Append
}

# Função de backup
function Backup-Files {
    param(
        [string]$SourceDir,
        [string]$DestinationDir
    )

    Write-Log "Iniciando backup de $SourceDir"

    try {
        if (-not (Test-Path $DestinationDir)) {
            New-Item -ItemType Directory -Path $DestinationDir -Force | Out-Null
        }

        $backupName = "Backup_$(Get-Date -Format 'yyyyMMdd')"
        $backupPath = Join-Path $DestinationDir $backupName

        # Compactar diretório
        Compress-Archive -Path $SourceDir -DestinationPath "$backupPath.zip" -Force

        Write-Log "Backup criado: $backupPath.zip"
        return $true
    }
    catch {
        Write-Log "Erro no backup: $_" -Level "ERROR"
        return $false
    }
}

# Função de monitoramento
function Monitor-System {
    Write-Log "Iniciando monitoramento do sistema"

    # CPU e Memória
    $cpu = Get-WmiObject Win32_Processor | Measure-Object -Property LoadPercentage -Average | Select-Object -ExpandProperty Average
    $memory = Get-WmiObject Win32_OperatingSystem | Select-Object @{Name="MemoryUsage";Expression={[math]::Round(($_.TotalVisibleMemorySize - $_.FreePhysicalMemory)/$_.TotalVisibleMemorySize*100, 2)}}

    Write-Log "CPU: $cpu% | Memória: $($memory.MemoryUsage)%"

    # Processos problemáticos
    $highCpuProcesses = Get-Process | Where-Object {$_.CPU -gt 50} | Select-Object Name, CPU, Id
    if ($highCpuProcesses) {
        Write-Log "Processos com alto uso de CPU:" -Level "WARNING"
        $highCpuProcesses | ForEach-Object {
            Write-Log "  - $($_.Name) (PID: $($_.Id)): $($_.CPU) segundos"
        }
    }

    # Espaço em disco
    $disks = Get-WmiObject Win32_LogicalDisk -Filter "DriveType=3"
    foreach ($disk in $disks) {
        $freePercent = [math]::Round(($disk.FreeSpace / $disk.Size) * 100, 2)
        if ($freePercent -lt 10) {
            Write-Log "Disco $($disk.DeviceID) com apenas $freePercent% livre" -Level "WARNING"
        }
    }

    # Serviços parados
    $stoppedServices = Get-Service | Where-Object {$_.Status -eq "Stopped" -and $_.StartType -eq "Automatic"}
    if ($stoppedServices) {
        Write-Log "Serviços automáticos parados:" -Level "WARNING"
        $stoppedServices | ForEach-Object {
            Write-Log "  - $($_.Name)"
        }
    }
}

# Função de limpeza
function Clean-TempFiles {
    Write-Log "Limpando arquivos temporários"

    $tempPaths = @(
        "$env:TEMP\*",
        "C:\Windows\Temp\*",
        "$env:LOCALAPPDATA\Temp\*"
    )

    $totalSize = 0
    foreach ($path in $tempPaths) {
        if (Test-Path $path) {
            $files = Get-ChildItem $path -File -Recurse -ErrorAction SilentlyContinue | Where-Object {$_.LastWriteTime -lt (Get-Date).AddDays(-7)}

            foreach ($file in $files) {
                try {
                    $totalSize += $file.Length
                    Remove-Item $file.FullName -Force -ErrorAction Stop
                }
                catch {
                    Write-Log "Não foi possível excluir $($file.FullName)" -Level "WARNING"
                }
            }
        }
    }

    $sizeMB = [math]::Round($totalSize / 1MB, 2)
    Write-Log "Limpados $sizeMB MB de arquivos temporários"
}

# Monitorar logs em tempo real
function Monitor-Logs {
    param(
        [string]$LogPath = "C:\Logs\",
        [string]$Filter = "ERROR"
    )

    Write-Host "Monitorando logs em $LogPath para '$Filter'..." -ForegroundColor Yellow

    $watcher = New-Object System.IO.FileSystemWatcher
    $watcher.Path = $LogPath
    $watcher.Filter = "*.log"
    $watcher.IncludeSubdirectories = $true
    $watcher.EnableRaisingEvents = $true

    $action = {
        $path = $Event.SourceEventArgs.FullPath
        $changeType = $Event.SourceEventArgs.ChangeType
        $timestamp = Get-Date -Format "HH:mm:ss"

        Write-Host "[$timestamp] $changeType detectado em $path" -ForegroundColor Cyan

        # Verificar se contém erro
        $content = Get-Content $path -Tail 1
        if ($content -match $Filter) {
            Write-Host "ERRO DETECTADO: $content" -ForegroundColor Red -BackgroundColor Black
            # Enviar alerta, email, etc.
        }
    }

    Register-ObjectEvent $watcher "Created" -Action $action
    Register-ObjectEvent $watcher "Changed" -Action $action

    Write-Host "Pressione Ctrl+C para parar..." -ForegroundColor Green
    while ($true) { Start-Sleep 1 }
}

# Instalar múltiplos programas automaticamente
function Install-Applications {
    param(
        [hashtable]$Applications = @{
            "Chrome" = "https://dl.google.com/chrome/install/chrome_installer.exe"
            "VSCode" = "https://code.visualstudio.com/sha/download?build=stable&os=win32-x64"
            "7Zip"   = "https://www.7-zip.org/a/7z2201-x64.exe"
        }
    )

    $installDir = "C:\Instaladores"
    if (-not (Test-Path $installDir)) {
        New-Item -ItemType Directory -Path $installDir | Out-Null
    }

    foreach ($app in $Applications.GetEnumerator()) {
        Write-Host "Instalando $($app.Name)..." -ForegroundColor Yellow

        $installerPath = Join-Path $installDir "$($app.Name).exe"

        # Download
        Write-Host "  Baixando..."
        Invoke-WebRequest -Uri $app.Value -OutFile $installerPath

        # Instalar
        Write-Host "  Executando instalador..."
        Start-Process $installerPath -ArgumentList "/S" -Wait

        Write-Host "  $($app.Name) instalado!" -ForegroundColor Green
    }

    Write-Host "Todas aplicações instaladas!" -ForegroundColor Green
}

# Criar múltiplos usuários a partir de CSV
function Create-UsersFromCSV {
    param(
        [string]$CsvPath = "usuarios.csv"
    )

    $users = Import-Csv $CsvPath

    foreach ($user in $users) {
        try {
            # Verificar se usuário já existe
            $existingUser = Get-LocalUser -Name $user.Username -ErrorAction SilentlyContinue

            if ($existingUser) {
                Write-Warning "Usuário $($user.Username) já existe. Atualizando..."
                Set-LocalUser -Name $user.Username -Password (ConvertTo-SecureString $user.Password -AsPlainText -Force)
            }
            else {
                # Criar novo usuário
                New-LocalUser -Name $user.Username `
                    -Password (ConvertTo-SecureString $user.Password -AsPlainText -Force) `
                    -FullName $user.FullName `
                    -Description $user.Description `
                    -PasswordNeverExpires:($user.PasswordNeverExpires -eq "Sim")

                Write-Host "Usuário $($user.Username) criado!" -ForegroundColor Green
            }

            # Adicionar ao grupo
            if ($user.Group) {
                Add-LocalGroupMember -Group $user.Group -Member $user.Username
                Write-Host "  Adicionado ao grupo: $($user.Group)" -ForegroundColor Cyan
            }
        }
        catch {
            Write-Error "Erro ao processar usuário $($user.Username): $_"
        }
    }
}
