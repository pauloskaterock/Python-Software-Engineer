🎯 EXEMPLO PRÁTICO: SISTEMA DE RPA COM WSL
bash
#!/bin/bash
# rpa_automation.sh - Sistema de RPA usando WSL

# Configurações
LOG_DIR="/var/log/rpa"
BACKUP_DIR="/mnt/c/Backups"
DATA_HOJE=$(date +%Y%m%d)
LOG_FILE="$LOG_DIR/rpa_$DATA_HOJE.log"

# Cores para output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# Função de logging
log() {
    local LEVEL=$1
    local MESSAGE=$2
    local TIMESTAMP=$(date '+%Y-%m-%d %H:%M:%S')

    case $LEVEL in
        "INFO")
            echo -e "${GREEN}[INFO]${NC} $MESSAGE"
            ;;
        "WARNING")
            echo -e "${YELLOW}[WARNING]${NC} $MESSAGE"
            ;;
        "ERROR")
            echo -e "${RED}[ERROR]${NC} $MESSAGE"
            ;;
    esac

    echo "[$TIMESTAMP] [$LEVEL] $MESSAGE" >> "$LOG_FILE"
}

# Criar diretórios se não existirem
setup_directories() {
    log "INFO" "Configurando diretórios..."

    sudo mkdir -p "$LOG_DIR"
    sudo mkdir -p "$BACKUP_DIR"

    # Dar permissão ao usuário atual
    sudo chown -R $USER:$USER "$LOG_DIR"
    sudo chown -R $USER:$USER "$BACKUP_DIR"
}

# Monitorar sistema
monitor_system() {
    log "INFO" "Iniciando monitoramento do sistema..."

    # CPU e Memória
    CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
    MEM_USAGE=$(free | grep Mem | awk '{print $3/$2 * 100.0}')

    log "INFO" "Uso de CPU: ${CPU_USAGE}%"
    log "INFO" "Uso de Memória: ${MEM_USAGE}%"

    if (( $(echo "$CPU_USAGE > 80" | bc -l) )); then
        log "WARNING" "Alto uso de CPU detectado!"
    fi

    # Espaço em disco
    DISK_USAGE=$(df / --output=pcent | tail -1 | tr -d '% ')
    if [ "$DISK_USAGE" -gt 80 ]; then
        log "WARNING" "Disco principal com ${DISK_USAGE}% de uso"
    fi

    # Processos problemáticos
    HIGH_CPU_PROCESSES=$(ps aux --sort=-%cpu | head -6 | tail -5)
    log "INFO" "Processos com maior uso de CPU:\n$HIGH_CPU_PROCESSES"
}

# Backup automático
backup_files() {
    local SOURCE_DIR=$1
    local BACKUP_NAME=$2

    log "INFO" "Iniciando backup de $SOURCE_DIR..."

    if [ ! -d "$SOURCE_DIR" ]; then
        log "ERROR" "Diretório fonte não existe: $SOURCE_DIR"
        return 1
    fi

    local BACKUP_PATH="$BACKUP_DIR/$BACKUP_NAME-$DATA_HOJE.tar.gz"

    # Compactar diretório
    tar -czf "$BACKUP_PATH" -C "$(dirname "$SOURCE_DIR")" "$(basename "$SOURCE_DIR")"

    if [ $? -eq 0 ]; then
        local SIZE=$(du -h "$BACKUP_PATH" | cut -f1)
        log "INFO" "Backup criado: $BACKUP_PATH ($SIZE)"
        return 0
    else
        log "ERROR" "Falha ao criar backup"
        return 1
    fi
}

# Limpar arquivos temporários
clean_temp_files() {
    log "INFO" "Limpando arquivos temporários..."

    # Limpar cache do apt
    sudo apt autoremove -y
    sudo apt autoclean

    # Limpar arquivos temporários do sistema
    find /tmp -type f -atime +7 -delete 2>/dev/null
    find /var/tmp -type f -atime +7 -delete 2>/dev/null

    # Limpar logs antigos
    find /var/log -name "*.log" -type f -mtime +30 -delete 2>/dev/null

    log "INFO" "Limpeza concluída"
}

# Processar arquivos CSV (exemplo de automação de dados)
process_csv_files() {
    local INPUT_DIR="/mnt/c/Data/Input"
    local OUTPUT_DIR="/mnt/c/Data/Output"

    log "INFO" "Processando arquivos CSV..."

    mkdir -p "$OUTPUT_DIR"

    for CSV_FILE in "$INPUT_DIR"/*.csv; do
        if [ -f "$CSV_FILE" ]; then
            local FILENAME=$(basename "$CSV_FILE")
            local OUTPUT_FILE="$OUTPUT_DIR/processed_$FILENAME"

            # Exemplo: contar linhas, extrair colunas, etc.
            TOTAL_LINES=$(wc -l < "$CSV_FILE")
            log "INFO" "Processando $FILENAME ($TOTAL_LINES linhas)"

            # Converter para UTF-8 se necessário
            iconv -f ISO-8859-1 -t UTF-8 "$CSV_FILE" -o "$OUTPUT_FILE" 2>/dev/null || cp "$CSV_FILE" "$OUTPUT_FILE"

            # Adicionar timestamp de processamento
            echo "# Processado em: $(date)" >> "$OUTPUT_FILE"

            log "INFO" "Arquivo processado: $OUTPUT_FILE"
        fi
    done
}

# Integração com Windows
windows_integration() {
    log "INFO" "Executando tarefas no Windows..."

    # Abrir relatório no Excel
    if [ -f "/mnt/c/Reports/relatorio.xlsx" ]; then
        log "INFO" "Abrindo relatório no Excel..."
        cmd.exe /c start excel.exe "C:\Reports\relatorio.xlsx" &
    fi

    # Enviar notificação Windows
    powershell.exe -Command "New-BurntToastNotification -Text 'RPA Concluído', 'Processo automático finalizado com sucesso'"

    # Copiar logs para Windows
    cp "$LOG_FILE" "/mnt/c/Logs/RPA/"
}

# Função principal
main() {
    log "INFO" "========================================"
    log "INFO" "SISTEMA DE RPA - WSL AUTOMATION"
    log "INFO" "Início: $(date)"
    log "INFO" "========================================"

    # Configurar ambiente
    setup_directories

    # Executar tarefas
    monitor_system
    backup_files "/home/$USER/Documents" "Documents_Backup"
    backup_files "/mnt/c/Important" "Windows_Backup"
    clean_temp_files
    process_csv_files
    windows_integration

    log "INFO" "========================================"
    log "INFO" "RPA CONCLUÍDO COM SUCESSO"
    log "INFO" "Fim: $(date)"
    log "INFO" "========================================"

    # Enviar resumo por email (opcional)
    # send_email_summary
}

# Tratamento de erros
trap 'log "ERROR" "Script interrompido pelo usuário"; exit 1' INT TERM
trap 'log "ERROR" "Erro na linha $LINENO"; exit 1' ERR

# Executar
main

MONITORAMENTO AUTOMÁTICO COM WSL:
bash
#!/bin/bash
# monitor_system.sh

# Configurações
ALERT_EMAIL="admin@empresa.com"
THRESHOLD_CPU=80
THRESHOLD_MEM=85
THRESHOLD_DISK=90

# Função de monitoramento
check_cpu() {
    local cpu_usage=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
    if (( $(echo "$cpu_usage > $THRESHOLD_CPU" | bc -l) )); then
        send_alert "CPU" "$cpu_usage%"
    fi
}

check_memory() {
    local mem_usage=$(free | grep Mem | awk '{print $3/$2 * 100.0}')
    if (( $(echo "$mem_usage > $THRESHOLD_MEM" | bc -l) )); then
        send_alert "Memória" "$mem_usage%"
    fi
}

check_disk() {
    local disk_usage=$(df / --output=pcent | tail -1 | tr -d '% ')
    if [ "$disk_usage" -gt "$THRESHOLD_DISK" ]; then
        send_alert "Disco" "$disk_usage%"
    fi
}

send_alert() {
    local resource=$1
    local usage=$2
    local message="ALERTA: $resource com uso elevado: $usage"

    echo "$message" | mail -s "Alerta Sistema WSL" "$ALERT_EMAIL"
    echo "[$(date)] $message" >> /var/log/system_alerts.log

    # Notificação Windows
    powershell.exe -Command "New-BurntToastNotification -Text 'Alerta Sistema', '$resource: $usage'"
}

# Executar verificações
while true; do
    check_cpu
    check_memory
    check_disk
    sleep 300  # Verificar a cada 5 minutos
done

EXEMPLO DE DEPLOYMENT AUTOMÁTICO:
bash
#!/bin/bash
# deploy_application.sh

# Configurações
APP_NAME="meu_app_rpa"
APP_DIR="/opt/$APP_NAME"
BACKUP_DIR="/mnt/d/Backups/$APP_NAME"
REPO_URL="https://github.com/seu_usuario/rpa_app.git"

# Criar backup antes de atualizar
create_backup() {
    echo "Criando backup..."
    local backup_file="$BACKUP_DIR/backup_$(date +%Y%m%d_%H%M%S).tar.gz"
    tar -czf "$backup_file" -C "$APP_DIR" .
    echo "Backup criado: $backup_file"
}

# Deploy da aplicação
deploy_app() {
    echo "Iniciando deploy..."

    if [ ! -d "$APP_DIR" ]; then
        echo "Clonando repositório..."
        sudo git clone "$REPO_URL" "$APP_DIR"
    else
        echo "Atualizando repositório..."
        cd "$APP_DIR"
        sudo git pull origin main
    fi

    # Instalar dependências
    echo "Instalando dependências..."
    cd "$APP_DIR"
    sudo pip3 install -r requirements.txt

    # Configurar serviços
    echo "Configurando serviços..."
    sudo cp "$APP_DIR/config/$APP_NAME.service" /etc/systemd/system/
    sudo systemctl daemon-reload
    sudo systemctl enable "$APP_NAME.service"
    sudo systemctl restart "$APP_NAME.service"

    echo "Deploy concluído!"
}

# Função principal
main() {
    create_backup
    deploy_app

    # Verificar status
    sudo systemctl status "$APP_NAME.service"
}

# Executar
main
⚠️ CUIDADOS IMPORTANTES:
Permissões de arquivos entre Windows e WSL são diferentes

Performance do sistema de arquivos cruzado pode ser lenta

Problemas de encoding (UTF-8 vs Windows-1252)

Fim de linha (CRLF vs LF)
