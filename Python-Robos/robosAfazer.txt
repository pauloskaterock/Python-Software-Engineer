class InvoiceProcessor:
    def __init__(self):

    def extract_invoice_data(self, pdf_path):
        """Extrai dados de PDF usando OCR"""
        # Converte PDF para imagem
        images = convert_from_path(pdf_path)
        text = ""

        for image in images:
            text += pytesseract.image_to_string(image, lang='por')

        # Extrai informações específicas (simplificado)

        return {
            'invoice_number': self.extract_invoice_number(text),
            'value': self.extract_value(text),

        }

    def process_all_invoices(self):
        """Processa todas as notas na pasta"""
        for file in os.listdir(self.input_folder):

🏆 VANTAGENS DO PYTHON PARA RPA
✅ Custo zero (open source)
✅ Extensibilidade ilimitada (qualquer biblioteca)
✅ Integração fácil com APIs, bancos de dados, sistemas
✅ Machine Learning nativo (scikit-learn, TensorFlow)
✅ Comunidade enorme (resolução rápida de problemas)
✅ Multiplataforma (Windows, Linux, Mac)

⚠️ DESAFIOS
❌ Manutenção de seletores (UI pode mudar)
❌ Processos muito visuais podem ser complexos
❌ Falta de ferramentas low-code para business users
❌ Orquestração requer setup adicional
❌ Monitoring/Alerting precisa ser construído
